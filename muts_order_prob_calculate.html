We classified 31 targeted genes into the pathways in the same functional lineage of the interaction of the multiple genes. Nucleophosmin gene were NPM1.

Splicing genes SRSF2, U2AF1, and SF3B1.

# Bootstrop Function
# Bishop, CM. Pattern recognition and machine learning. Springer. 23, 656 (2006).
def MkLst(input_df):
  """
  For a given data frame, return gene list that we need for the bootstrap
  """
  gene_lst = []
  for i in range(len(input_df)):
    gene_ = input_df["gene"][i]
    frequency_ = input_df["frequency"][i]

    for _ in range(frequency_):
      gene_lst.append(gene_)

  return gene_lst

def BootStrap(input_df, n_iter=1000):
  """
  This function does bootstrap methods for mutation order probability
  :panam input_df: this is input dataframe 
  :param n_iter: number of random sampling with replacement
  """
  input_lst = MkLst(input_df)
  prob_dic = defaultdict(float)
  freaq_dic = defaultdict(float)
  prob_df = pd.DataFrame()
  frequency_lst = []
  
  for i in tqdm(range(n_iter)):
    resamples =  resample(input_lst, replace=True, n_samples=len(input_lst), random_state=i)
    cnt_resamples = Counter(resamples)

    for k_, v_ in cnt_resamples.items():
      n_samples_ = float(Fraction(v_)/Fraction(len(input_lst)))
      prob_dic[k_] += n_samples_
      freaq_dic[k_] += v_
  
  for k_ in prob_dic.keys():
    prob_dic[k_] =  float(Fraction(prob_dic[k_])/Fraction(n_iter))
    freaq_dic[k_] = float(Fraction(freaq_dic[k_])/Fraction(n_iter))

  prob_df = pd.DataFrame({"gene":list(prob_dic.keys()), "probability":list(prob_dic.values())})
  
  prob_df.sort_values(by="probability", ascending=False, inplace=True)
  prob_df.reset_index(drop=True, inplace=True)

  for j in range(len(prob_df)):
    gene_ = prob_df.iloc[:,0][j]
    frequency_lst.append(freaq_dic[gene_])

  prob_df["frequency"] = frequency_lst
  prob_df["frequency"] = prob_df["frequency"].round().astype(int)

  return prob_df

# Function to calculate the 95% confidence interval
# The 95% confidence intervals (CI) was calculated using the Sison and Glaz method in the MultinomCI of R package.
# This is the mathematical details are in Theorem 2.1.(Sison, C.P and Glaz, J., 1995).
# Sison, C.P and Glaz, J. Simultaneous confidence intervals and sample size determination for multinomial proportions. Journal of the American Statistical Association 90, 366-369 (1995).
# http://finzi.psych.upenn.edu/library/DescTools/html/MultinomCI.html
def multinomialCI(x, probs=None, conf=0.95):
  """
  This function is translated to python from R
  https://github.com/cran/MultinomialCI/blob/master/R/sison.R
  """
  x = np.array(x)
  n = sum(x)
  k = len(x)
  c = 0
  pold = 0
  
  for cc in range(1, n+1):
   poi = _truncpoi(cc, x, n, k)
   if (poi > conf) & (pold < conf): 
     c = cc 
     break 
   pold=poi
   
  salida = np.zeros((k, 2))
  delta = (conf-pold)/(poi-pold)
  out = np.zeros((k, 5))
  num = np.zeros((k, 1))
  c -= 1  
  vol1 = 1
  vol2 = 1
  
  for i in range(k):
   num[i, 0] = i
   if probs is None:
     obsp = x[i]/n
   else:
     obsp = probs[i]

   out[i, 0] = obsp
   out[i, 1] = obsp-c/n
   out[i, 2] = obsp+c/n+2*delta/n

   if out[i, 1] < 0: out[i, 1] = 0
   if out[i, 2] > 1: out[i, 2] = 1
   out[i, 3] = obsp-c/n-1/n
   out[i, 4] = obsp+c/n+1/n

   if out[i, 1] < 0: out[i, 1] = 0
   if out[i, 2] > 1: out[i, 2] = 1
   vol1 *= (out[i, 2]-out[i, 1])
   vol2 *= (out[i, 4]-out[i, 3])
  
   salida[i, 0] = out[i, 1]
   salida[i, 1] = out[i, 2]
  
  cov=100*(conf)
  sg=(x+delta)/n

  return salida

def _moments(c, l):
  """
  This function is translated to python from R
  https://github.com/cran/MultinomialCI/blob/master/R/aux-fn.R
  """
  a = l + c
  b = l - c
  if b<0: b=0
  if b>0: den=poisson.cdf(a, l) - poisson.cdf(b-1, l)
  if b==0: den=poisson.cdf(a, l)
  
  mu=np.zeros((4, 1))
  mom=np.zeros((5, 1))

  for r in range(1, 5):
    poisA=0
    poisB=0
    if (a-r) >= 0: poisA = poisson.cdf(a, l) - poisson.cdf(a-r, l)
    if (a-r) < 0: poisA = poisson.cdf(a, l)
    if (b-r-1) >=0: poisB = poisson.cdf(b-1, l) - poisson.cdf(b-r-1, l)
    if ((b-r-1) < 0) & ((b-1) >= 0): poisB = poisson.cdf(b-1, l)
    if ((b-r-1) < 0) & ((b-1) < 0): poisB = 0

    mu[r-1, 0] = (l**r)*(1-(poisA-poisB)/den)
   
  mom[0] = mu[0]
  mom[1] = mu[1]+mu[0]-mu[0]**2
  mom[2] = mu[2]+mu[1]*(3-3*mu[0])+(mu[0]-3*mu[0]**2+2*mu[0]**3)
  mom[3] = mu[3]+mu[2]*(6-4*mu[0])+mu[1]*(7-12*mu[0]+6*mu[0]**2)+mu[0]-4*mu[0]**2+6*mu[0]**3-3*mu[0]**4
  mom[4] = den

  return mom

def _truncpoi(c, x, n, k):
  """
  This function is translated to python from R
  https://github.com/cran/MultinomialCI/blob/master/R/aux-fn.R
  """
  m = np.zeros((k, 5))
  for i in range(k):
    l = x[i]
    mom = _moments(c, l)
    
    for j in range(5):
      m[i, j] = mom[j]
    
  for i in range(k):
    m[i, 3] = m[i, 3]-3*m[i, 1]**2
     
  s=np.sum(m, axis=0)
  s1=s[0]
  s2=s[1]
  s3=s[2]
  s4=s[3]

  probn = 1/(poisson.cdf(n, n)-poisson.cdf(n-1,n))
  z = (n-s1)/sqrt(s2)
  g1 = s3/(s2**(3/2))
  g2 = s4/(s2**2)
  poly = 1+g1*(z**3-3*z)/6+g2*(z**4-6*z**2+3)/24+g1**2*(z**6-15*z**4+45*z**2-15)/72
  f = poly*np.exp(-z**2/2)/(sqrt(2)*gamma(0.5))
    
  probx=1
  for i in range(k):
    probx = probx*m[i, 4]
   
  return (probn*probx*f/sqrt(s2))
